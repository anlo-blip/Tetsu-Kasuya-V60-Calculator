        'acidity-high': [0.65, 0.35]
      };
      const [p1, p2] = splits[balance] || [0.5, 0.5];
      return { pour1: Math.round(first40 * p1), pour2: Math.round(first40 * p2) };
    }

    function calculateTemperature(process, roast) {
      let temp = 92;
      if (roast === 'light') temp += 1;
      if (process === 'washed') temp += 1;
      if (process === 'natural') temp -= 1;
      return temp;
    }

    function calculateGrindSize(ratio, body, roast, process) {
      let grind = 53;
      if (ratio >= 16) grind += 2;
      if (body === 'light') grind -= 2;
      if (body === 'heavy') grind += 2;
      if (roast === 'light') grind -= 2;
      if (process === 'natural') grind += 3;
      if (process === 'washed') grind -= 2;
      return Math.max(45, Math.min(60, grind));
    }

    function calculateRPM(process, body) {
      let rpm = 100;
      if (process === 'natural') rpm -= 15;
      if (process === 'washed') rpm += 10;
      if (body === 'light') rpm += 10;
      if (body === 'heavy') rpm -= 10;
      return rpm;
    }

    function calculateTargetTime(roast, process, body, grind) {
      let min = 2, sec = 30;
      if (roast === 'light') sec += 15;
      if (process === 'natural') sec += 10;
      if (body === 'heavy') sec += 10;
      if (grind < 50) sec -= 10;
      if (grind > 55) sec += 10;
      if (sec >= 60) { min += Math.floor(sec / 60); sec = sec % 60; }
      const endSecRaw = sec + 20;
      const endMin = min + Math.floor(endSecRaw / 60);
      const endSec = endSecRaw % 60;
      return `${min}:${String(sec).padStart(2,'0')} – ${endMin}:${String(endSec).padStart(2,'0')}`;
    }

    function calculateRecipe() {
      const coffeeWeight = parseFloat(document.getElementById('coffeeWeight').value || 20);
      const ratio = parseFloat(document.getElementById('ratio').value || 15);
      const balance = document.getElementById('balance').value;
      const body = document.getElementById('body').value;
      const process = document.getElementById('process').value;
      const roast = document.getElementById('roast').value;

      const totalWater = Math.round(coffeeWeight * ratio);
      const first40 = totalWater * 0.4;
      const last60 = totalWater * 0.6;
      const split = calculatePourSplit(balance, first40);
      const numberOfPours = body === 'heavy' ? 4 : body === 'medium' ? 5 : 6;
      const poursInLast60 = numberOfPours - 2;
      const each = Math.round(last60 / poursInLast60);
      const timings = ['1:20','1:50','2:15','2:40'];

      const pours = [
        { number:1, water:split.pour1, time:'0:00', purposeKey:'acidity' },
        { number:2, water:split.pour2, time:'0:45', purposeKey:'sweetness' }
      ];

      for (let i = 0; i < poursInLast60; i++) {
        pours.push({
          number: i + 3,
          water: each,
          time: timings[i] || `${i + 2}:00`,
          purposeKey:'body'
        });
      }

      const temp = calculateTemperature(process, roast);
      const grindSize = calculateGrindSize(ratio, body, roast, process);
      const rpm = calculateRPM(process, body);
      const targetTime = calculateTargetTime(roast, process, body, grindSize);

      const data = { coffeeWeight, totalWater, ratio, temp, grindSize, rpm, targetTime, pours, balance, body, process, roast };
      safeSet('lastRecipe', JSON.stringify(data));
      displayResults(data);

      document.getElementById('results').classList.add('active');
      document.getElementById('results').scrollIntoView({ behavior:'smooth' });
    }

    function fallbackCopy(text) {
      const area = document.createElement('textarea');
      area.value = text;
      document.body.appendChild(area);
      area.select();
      try { document.execCommand('copy'); } catch(e) {}
      document.body.removeChild(area);
    }

    function copyRecipe() {
      const raw = safeGet('lastRecipe');
      if (!raw) return alert(translations[currentLang].needRecipeFirst);
      const data = JSON.parse(raw);

      const text = currentLang === 'ar'
        ? `وصفة V60\nالبن: ${data.coffeeWeight}g\nالماء: ${data.totalWater}ml\nالريشو: 1:${data.ratio}\nالحرارة: ${data.temp}°C\nالطحن: ${data.grindSize}\nالزمن: ${data.targetTime}`
        : `V60 Recipe\nCoffee: ${data.coffeeWeight}g\nWater: ${data.totalWater}ml\nRatio: 1:${data.ratio}\nTemp: ${data.temp}°C\nGrind: ${data.grindSize}\nTime: ${data.targetTime}`;

      if (navigator.clipboard && navigator.clipboard.writeText) {
        navigator.clipboard.writeText(text)
          .then(() => alert(translations[currentLang].copySuccess))
          .catch(() => { fallbackCopy(text); alert(translations[currentLang].copySuccess); });
      } else {
        fallbackCopy(text);
        alert(translations[currentLang].copySuccess);
      }
    }

    function downloadRecipe() {
      const raw = safeGet('lastRecipe');
      if (!raw) return alert(translations[currentLang].needRecipeFirst);
      const blob = new Blob([raw], { type:'application/json;charset=utf-8' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = 'tetsu-v60-recipe.json';
      a.click();
      URL.revokeObjectURL(url);
    }

    function saveFavorite() {
      const raw = safeGet('lastRecipe');
      if (!raw) return alert(translations[currentLang].needRecipeFirst);
      const data = JSON.parse(raw);

      let favorites = [];
      try { favorites = JSON.parse(safeGet('favoriteRecipes','[]')) || []; } catch(e) {}

      const exists = favorites.some(f =>
        f.coffeeWeight === data.coffeeWeight &&
        f.ratio === data.ratio &&
        f.balance === data.balance &&
        f.body === data.body &&
        f.process === data.process &&
        f.roast === data.roast
      );

      if (exists) return alert(translations[currentLang].favExists);

      favorites.push({
        id: Date.now(),
        title: `${data.coffeeWeight}g | 1:${data.ratio} | ${data.process} | ${data.roast}`,
        ...data
      });

      safeSet('favoriteRecipes', JSON.stringify(favorites));
      alert(translations[currentLang].favSaved);
      renderFavorites();
    }
